package player;

import token.Worker;

import java.util.ArrayList;
import java.util.List;

public class Player {
    private int score;
    private final String playername;
    private List<Worker> workers;

    /**
     * Initializes a new player instance with name.
     *
     * @param name Define the name of the player.
     */
    public Player(String name) {
        this.playername = name;
    }

    /**
     * Get the name of the player.
     *
     * @return The name of the player.
     */
    public String getName() {
        return this.playername;
    }

    /**
     * Initialized the workers for the player with two different names which represent worker1 and worker2.
     *
     * @param name1 The name of first worker.
     * @param name2 The name of second worker.
     * @return {@code true} if the input names are different.
     */
    public boolean initialWorkers(String name1, String name2) {
        if (name1 != name2) {
            this.workers = new ArrayList<>();
            Worker worker1 = new Worker(this.playername + name1);
            Worker worker2 = new Worker(this.playername + name2);
            workers.add(worker1);
            workers.add(worker2);
            System.out.println("Successfully initialize " + worker1.getName() + " and " + worker2.getName() + "!");
            return true;
        } else {
            System.out.println("Don not use the same name of two workers!");
            return false;
        }
    }

    /**
     * Get the list of workers of the player.
     *
     * @return The {@link Worker} workers of player class.
     */
    public List<Worker> getWorkers() {
        return this.workers;
    }

    public void setScore(int score){
        this.score = score;
    }

    public int getScore(){
        return this.score;
    }

}
